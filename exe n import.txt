def load_photo(path):
        if os.path.exists(path):
            try:
                img = Image.open(path)
                img.thumbnail((150, 150))
                img_tk = ImageTk.PhotoImage(img)
                photo_label.configure(image=img_tk, text="", bg="white")
                photo_label.image = img_tk
            except:
                photo_label.configure(text="Image Error", image="", bg="gray")

    def browse_photo():
        #messagebox.showinfo("","")
        file_path = filedialog.askopenfilename(filetypes=[("Image Files", "*.jpg *.jpeg *.png")])
        if file_path:
            # Convert to relative path for storage
            base_dir = os.path.dirname(os.path.abspath(__file__))
            try:
                relative_path = os.path.relpath(file_path, base_dir)
                photo_path.delete(0, tk.END)
                photo_path.insert(0, relative_path)
            except ValueError:
                photo_path.delete(0, tk.END)
                photo_path.insert(0, file_path)
            
            load_photo(file_path)  # Load with original absolute path
    
    tk.Button(photo_frame, text="Browse Photo", command=browse_photo).pack(pady=5)
